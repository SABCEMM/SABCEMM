80  static double mean(std::vector<double>& v){
81  double sum = std::accumulate(v.begin(), v.end(), 0.0);
82  double mean = sum / static_cast<double>(v.size());
83  return mean;
84  }
85  static double std(std::vector<double>& v){
86  double mean_ = mean(v);
87  std::vector<double> diff(v.size());
88  std::transform(v.begin(), v.end(), diff.begin(), [mean_](double x) { return x - mean_; });
89  double sq_sum = std::inner_product(diff.begin(), diff.end(), diff.begin(), 0.0);
90  double var = std::sqrt(sq_sum / static_cast<double>(v.size()));
91  return var;
92  }
